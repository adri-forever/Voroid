cmake_minimum_required(VERSION 3.10)

project(Voroid VERSION 1.0)

set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

include_directories(
    "thirdparty/glm-1.0.1"
    "thirdparty"
    "thirdparty/glad/include"
    "include"
    "include/utils"
)

# Bad way to link all cpp
file(GLOB_RECURSE SOURCES CONFIGURE_DEPENDS "src/*.cpp")
# Add the GL loader
list(APPEND SOURCES "thirdparty/glad/src/glad.c")
add_executable(Voroid ${SOURCES})

find_package(SDL3 REQUIRED)
target_link_libraries(Voroid PRIVATE SDL3::SDL3)

# Remove already present assets
add_custom_command(TARGET Voroid POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E remove_directory
        ${CMAKE_BINARY_DIR}/assets
)
# Copy the assets to the build directory
add_custom_command(TARGET Voroid POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
        ${CMAKE_SOURCE_DIR}/assets
        ${CMAKE_BINARY_DIR}/assets
)
